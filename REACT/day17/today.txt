const App = () =>{

    const [form, setForm] = useState("")
    const handleChange = (e)=>{
        const data = e.target.value
        //setForm(data)

        if(data>18){
            setForm(data)
        }else{
            setForm("Not Eligible")
        }
    }

    return(
        <>
        <div>
        <h1Form Handling</h1>
        <input tags>

        <input onChange = {handleChange} />
        </div>
        </>

    )
}
